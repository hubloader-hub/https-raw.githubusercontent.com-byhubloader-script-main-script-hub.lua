local dragging = false
local dragInputId = nil
local dragStartPos, startGuiPos

top.InputBegan:Connect(function(input, gp)
    if gp then return end
    local isTouch = input.UserInputType == Enum.UserInputType.Touch
    local isMouse = input.UserInputType == Enum.UserInputType.MouseButton1
    if (isTouch or isMouse) and not dragging then
        dragging = true
        dragInputId = input
        dragStartPos = input.Position
        startGuiPos = main.Position
        local conn
        conn = input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                if dragInputId == input then
                    dragging = false
                    dragInputId = nil
                end
                conn:Disconnect()
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if not dragging or input ~= dragInputId then return end
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        local delta = input.Position - dragStartPos
        main.Position = UDim2.new(startGuiPos.X.Scale, startGuiPos.X.Offset + delta.X, startGuiPos.Y.Scale, startGuiPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input == dragInputId then
        dragging = false
        dragInputId = nil
    end
end)

local function executeScript(scriptSource)
    if not scriptSource or scriptSource == "" then return end
    local f, err = loadstring(scriptSource)
    if f then
        local ok, runErr = pcall(f)
        if not ok then warn("[HubLoader] Runtime: " .. tostring(runErr)) end
    else
        warn("[HubLoader] Syntax: " .. tostring(err))
    end
end
