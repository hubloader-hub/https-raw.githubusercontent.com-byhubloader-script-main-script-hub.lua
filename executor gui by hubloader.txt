local dragging = false
local dragInputId = nil
local dragStartPos, startGuiPos

top.InputBegan:Connect(function(input, gp)
    if gp then return end
    local isTouch = input.UserInputType == Enum.UserInputType.Touch
    local isMouse = input.UserInputType == Enum.UserInputType.MouseButton1
    if (isTouch or isMouse) and not dragging then
        dragging = true
        dragInputId = input
        dragStartPos = input.Position
        startGuiPos = main.Position
        local conn
        conn = input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                if dragInputId == input then
                    dragging = false
                    dragInputId = nil
                end
                conn:Disconnect()
            end
        end)
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if not dragging or input ~= dragInputId then return end
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        local delta = input.Position - dragStartPos
        main.Position = UDim2.new(startGuiPos.X.Scale, startGuiPos.X.Offset + delta.X, startGuiPos.Y.Scale, startGuiPos.Y.Offset + delta.Y)
    end
end)

UserInputService.InputEnded:Connect(function(input)
    if input == dragInputId then
        dragging = false
        dragInputId = nil
    end
end)

local function doMenuGlitch()
    if not main.Visible then return end
    local basePos = main.Position
    local elapsed = 0
    local conn
    conn = RunService.Heartbeat:Connect(function(dt)
        if elapsed >= 0.28 then
            for _, gl in ipairs(menuGlitchLines) do gl.BackgroundTransparency = 1 end
            menuRGB.BackgroundTransparency = 1
            main.Position = basePos
            conn:Disconnect()
            return
        end
        elapsed = elapsed + dt

        for _, gl in ipairs(menuGlitchLines) do
            gl.Position = UDim2.new(math.random(0,20)/100, 0, math.random(5,92)/100, 0)
            gl.Size = UDim2.new(math.random(15,95)/100, 0, 0, math.random(1,7))
            gl.BackgroundColor3 = Color3.fromRGB(math.random(0,255),math.random(0,255),math.random(0,255))
            gl.BackgroundTransparency = math.random(15,55)/100
        end

        menuRGB.BackgroundColor3 = Color3.fromRGB(math.random(0,255), 0, math.random(80,255))
        menuRGB.BackgroundTransparency = math.random(90,98)/100

        if math.random(1, 3) == 1 then
            main.Position = UDim2.new(basePos.X.Scale, basePos.X.Offset + math.random(-2,2), basePos.Y.Scale, basePos.Y.Offset + math.random(-1,1))
        else
            main.Position = basePos
        end
    end)
end
